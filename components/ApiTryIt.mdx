export function ApiTryIt({ operationId, title = "Try This Endpoint" }) {
  const baseUrl = "https://raw.githubusercontent.com/vamsinole/docs/refs/heads/main/api-reference/openapi.json";
  const swaggerUrl = `${baseUrl}#/${operationId}`;
  
  return (
    <div>
      <h3>{title}</h3>
      <p>Test this API endpoint directly:</p>
      <iframe 
        src={`https://petstore.swagger.io/?url=${baseUrl}#/${operationId}`}
        width="100%" 
        height="600px"
        style={{
          border: "none", 
          borderRadius: "8px", 
          boxShadow: "0 4px 6px rgba(0, 0, 0, 0.1)", 
          marginTop: "20px"
        }}
        title={`${operationId} API Playground`}
      />
    </div>
  );
}
